3// Einfache Funktionen fuer eine LED zum testen von Custom Simulink Bloecken



4#include "LED.h"



5#include "stm32f4xx.h"



6#include "stm32f4xx_rcc.h"



7#include "stm32f4xx_gpio.h"



8



9void initLED()



10{



11	  GPIO_InitTypeDef  GPIO_InitStructure;



12	  RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOD, ENABLE);



13	  // Config PD12 als Digital-Ausgang



14	  GPIO_InitStructure.GPIO_Pin 	= GPIO_Pin_12;



15	  GPIO_InitStructure.GPIO_Mode 	= GPIO_Mode_OUT;



16	  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;



17	  GPIO_InitStructure.GPIO_PuPd 	= GPIO_PuPd_UP;



18	  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;



19	  GPIO_Init(GPIOD, &GPIO_InitStructure);



20}



21void setLED(bool ledState)



22{



23	ledState ? GPIO_SetBits(GPIOD, GPIO_Pin_12) : GPIO_ResetBits(GPIOD, GPIO_Pin_12);



24}
